import mpmath as mp

# ضبط الدقة العشرية العالية
mp.dps = 30  # 30 رقمًا عشريًا

def zeta_partial_sum(s, N):
    """
    المجموع الجزئي لدالة زيتا: Σ_{n=1}^{N} n^{-s}
    s عدد مركب
    N عدد صحيح
    """
    total = mp.mpf(0)
    for n in range(1, N+1):
        total += mp.power(mp.mpf(n), -s)
    return total

def zeta_exact(s):
    """
    القيمة الدقيقة لدالة زيتا باستخدام mpmath
    """
    return mp.zeta(s)

def hypothetical_law(t):
    """
    القانون المفترض من النص: 1/sqrt((0.5)^2 + t^2)
    """
    sigma = mp.mpf(0.5)
    return 1 / mp.sqrt(sigma**2 + mp.mpf(t)**2)

python
import math
import cmath

def zeta_partial_sum(s_real, s_imag, N):
    """
    المجموع الجزئي لدالة زيتا: Σ_{n=1}^{N} n^{-s}
    s = s_real + i * s_imag
    N: عدد صحيح
    """
    total_real = 0.0
    total_imag = 0.0
    
    for n in range(1, N + 1):
        # n^(-s) = e^(-s * ln(n))
        ln_n = math.log(n)
        # الجزء الحقيقي: n^(-sigma) * cos(t * ln(n))
        # الجزء التخيلي: -n^(-sigma) * sin(t * ln(n))
        factor = math.exp(-s_real * ln_n)  # n^(-sigma)
        angle = -s_imag * ln_n  # -t * ln(n)
        
        real_part = factor * math.cos(angle)
        imag_part = factor * math.sin(angle)
        
        total_real += real_part
        total_imag += imag_part
    
    return total_real, total_imag

def magnitude(x, y):
    """حساب طول المتجه"""
    return math.sqrt(x*x + y*y)

def hypothetical_law(t):
    """
    القانون المفترض: 1/sqrt(0.25 + t^2)
    """
    return 1.0 / math.sqrt(0.25 + t*t)

import math

def vector_sum_and_magnitude(s_real, s_imag, N):
    """
    جمع اتجاهي عادي لـ n^{-s} من n=1 إلى N
    ثم إرجاع المقدار النهائي
    """
    total_real = 0.0
    total_imag = 0.0
    
    print(f"\n--- جمع اتجاهي لـ N = {N} ---")
    print(f"{'n':<4} {'المتجه (real, imag)':<25} {'المجموع التراكمي':<30} {'مقدار المجموع':<15}")
    print("-" * 80)
    
    for n in range(1, N + 1):
        ln_n = math.log(n)
        factor = math.exp(-s_real * ln_n)
        angle = -s_imag * ln_n
        
        real_part = factor * math.cos(angle)
        imag_part = factor * math.sin(angle)
        
        total_real += real_part
        total_imag += imag_part
        
        mag = math.sqrt(total_real**2 + total_imag**2)
        
        # اطبع أول 10 متجهات فقط لتجنب التكرار
        if n <= 10 or n == N:
            vec_str = f"({real_part:.4f}, {imag_part:.4f})"
            sum_str = f"({total_real:.4f}, {total_imag:.4f})"
            print(f"{n:<4} {vec_str:<25} {sum_str:<30} {mag:.6f}")
    
    return total_real, total_imag, math.sqrt(total_real**2 + total_imag**2)

    is_prime = [True] * (N + 1)
    primes = []
    
    # تخزين العوامل الأولية
    smallest_prime = [0] * (N + 1)

        if is_prime[i]:
            primes.append(i)
            smallest_prime[i] = i
            mu[i] = -1  # الأعداد الأولية لها μ = -1

import mpmath as mp
import math
import matplotlib.pyplot as plt
import numpy as np

# ضبط الدقة العالية
mp.dps = 50

# الثابت الكوني
COSMIC = mp.pi / mp.sqrt(8)  # 1.0472...

def compute_mobius_upto(N):
    """
    حساب دالة موبيوس μ(n) لكل n من 1 إلى N باستخدام منخل (Sieve)
    """
    mu = [1] * (N + 1)
    is_prime = [True] * (N + 1)
    primes = []
    
    for i in range(2, N + 1):
        if is_prime[i]:
            primes.append(i)
            mu[i] = -1
        
        for p in primes:
            if i * p > N:
                break
            is_prime[i * p] = False
            
            if i % p == 0:
                mu[i * p] = 0
                break
            else:
                mu[i * p] = -mu[i]
    
    return mu

def mobius(n, mu_array):
    """إرجاع μ(n) من المصفوفة المحسوبة مسبقاً"""
    if n == 1:
        return 1
    if n < len(mu_array):
        return mu_array[n]
    return 0

def compute_pulse(rho, N, mu_array):
    """حساب النبضة M_N(rho)"""
    M_N = mp.mpc(0)
    for n in range(1, N+1):
        mu_n = mobius(n, mu_array)
        if mu_n != 0:
            M_N += mu_n * (mp.mpf(n) ** (-rho))
    return abs(M_N)

def analyze_convergence(zero_index, N_values, mu_array):
    """
    تحليل التقارب لصفر معين
    """
    rho = mp.zetazero(zero_index)
    t = float(rho.imag)
    zeta_prime = abs(mp.zeta(rho, derivative=1))
    
    print(f"\n{'='*70}")
    print(f"تحليل الصفر Z{zero_index}: t = {t:.6f}, |ζ'(ρ)| = {float(zeta_prime):.6f}")
    print(f"{'='*70}")
    
    results = []
    
    print(f"{'N':<12} {'|M_N|':<15} {'Pulse Value':<15} {'Ratio':<12} {'Error %':<12} {'Convergence Rate':<20}")
    print("-" * 95)
    
    prev_error = None
    
    for N in N_values:
        abs_M = compute_pulse(rho, N, mu_array)
        pulse = (abs_M * zeta_prime) / mp.log(N)
        ratio = pulse / COSMIC
        error = abs(1 - float(ratio)) * 100
        
        # حساب سرعة التقارب (نسبة تحسن الخطأ)
        convergence_rate = ""
        if prev_error is not None and prev_error > 0:
            rate = prev_error / error if error > 0 else float('inf')
            if rate > 1.1:
                convergence_rate = f"تحسن بنسبة {rate:.2f}x"
            elif rate < 0.9:
                convergence_rate = "تباطؤ"
            else:
                convergence_rate = "ثابت"
        
        print(f"{N:<12} {float(abs_M):<15.6f} {float(pulse):<15.6f} {float(ratio):<12.6f} {error:<12.4f} {convergence_rate}")
        
        results.append({
            'N': N,
            'abs_M': float(abs_M),
            'pulse': float(pulse),
            'ratio': float(ratio),
            'error': error
        })
        
        prev_error = error
    
    return results, t, float(zeta_prime)

def plot_convergence(all_results):
    """
    رسم بياني لتحليل التقارب
